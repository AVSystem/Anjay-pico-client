cmake_minimum_required(VERSION 3.13)

add_executable(secure_communication
               main.c
               )

target_link_libraries(secure_communication
                      pico_stdlib
                      anjay-pico
                      FreeRTOS
                      )

target_include_directories(secure_communication PRIVATE
                           ${COMMON_DIR}/config
                           )

target_compile_definitions(secure_communication PRIVATE
                           WIFI_SSID=\"${WIFI_SSID}\"
                           WIFI_PASSWORD=\"${WIFI_PASSWORD}\"
                           ENDPOINT_NAME=\"${ENDPOINT_NAME}\"
                           PSK_IDENTITY=\"${PSK_IDENTITY}\"
                           PSK_KEY=\"${PSK_KEY}\"
                           )

pico_enable_stdio_usb(secure_communication 1)
pico_enable_stdio_uart(secure_communication 0)

pico_add_extra_outputs(secure_communication)
